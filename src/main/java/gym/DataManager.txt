package gym;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;
import java.io.File;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

public class DataManager {
    private final File file;
    private final ObjectMapper mapper;

    /**
     * Initializes the DataManager with a target JSON file path.
     * 
     * @param filename The path to the JSON file (e.g., "customers.json").
     */
    public DataManager(String filename) {
        if (filename == null || filename.isBlank()) {
            throw new IllegalArgumentException("Filename cannot be null or empty.");
        }
        this.file = new File(filename);
        this.mapper = new ObjectMapper();
        
        // Explicitly registers support for Java 8 Date/Time types like LocalDate
        this.mapper.registerModule(new JavaTimeModule());
    }

    /**
     * Loads a JSON array from the file and maps it directly into a typed ArrayList.
     * If the file does not exist yet or is empty, it gracefully returns an empty list.
     * 
     * @param clazz The type of class to map the JSON elements into.
     * @return An ArrayList containing the parsed objects.
     * @throws IOException If there is an issue accessing or parsing the file.
     */
    public <T> ArrayList<T> loadData(Class<T> clazz) throws IOException {
        if (!file.exists() || file.length() == 0) {
            return new ArrayList<>();
        }
        
        return mapper.readValue(
            file, 
            mapper.getTypeFactory().constructCollectionType(ArrayList.class, clazz)
        );
    }

    /**
     * Saves any collection of objects cleanly as a pretty-printed JSON array.
     * 
     * @param data The List of objects to serialize and save.
     * @throws IOException If there is an issue writing to the file.
     */
    public void saveData(List<?> data) throws IOException {
        mapper.writerWithDefaultPrettyPrinter().writeValue(file, data);
    }
}
